    <script>
        // Función para actualizar la pantalla de la calculadora
        function pressKey(key) {
            document.getElementById("resultado").value += key;
        }

        // Función para realizar el cálculo
        function calculate() {
            let expression = document.getElementById("resultado").value;
            try {
                let result = eval(expression);
                document.getElementById("resultado").value = result;
            } catch (e) {
                document.getElementById("resultado").value = "Error";
            }
        }

        // Función para limpiar la pantalla
        function clearScreen() {
            document.getElementById("resultado").value = "";
        }

        // Función para borrar el último carácter
        function backspace() {
            let currentInput = document.getElementById("resultado").value;
            document.getElementById("resultado").value = currentInput.slice(0, -1);
        }
    </script>
